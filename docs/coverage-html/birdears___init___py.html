262class _GetchUnix: 

263 def __init__(self): 

264 import tty 

265 import sys 

266 

267 def __call__(self): 

268 import sys 

269 import tty 

270 import termios 

271 fd = sys.stdin.fileno() 

272 old_settings = termios.tcgetattr(fd) 

273 try: 

274 tty.setraw(sys.stdin.fileno()) 

275 ch = sys.stdin.read(1) 

276 finally: 

277 termios.tcsetattr(fd, termios.TCSADRAIN, old_settings)
